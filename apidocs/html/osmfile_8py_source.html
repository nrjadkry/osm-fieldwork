356 
357  Returns:
358  (list): The entries in the OSM XML file

362  xml = file.read(size) # Instances are small, read the whole file
363  doc = xmltodict.parse(xml)
364  if "osm" not in doc:
365  logging.warning("No data in this instance")
366  return False
367  field = doc["osm"]
368 
369  if "node" not in field:
370  logging.warning("No nodes in this instance")
371  return False
372 
373  if type(field["node"]) == dict:
374  attrs = dict()
375  tags = dict()
376  for k, v in field["node"].items():
377  if k[0] == '@':
378  attrs[k[1:]] = v
379  else:
380  if type(field["node"]['tag']) == dict:
381  tags[field["node"]['tag']["@k"]] = field["node"]['tag']["@v"].strip()
382  else:
383  for pair in field['node']['tag']:
384  tags[pair['@k']] = pair['@v']
385 
386  node = {"attrs": attrs, "tags": tags}
387  self.datadata[node["attrs"]["id"]] = node
388  else:
389  for node in field["node"]:
390  attrs = {
391  "id": int(node["@id"]),
392  "lat": node["@lat"][:10],
393  "lon": node["@lon"][:10],
394  }
395  if '@timestamp' in node:
396  attrs["timestamp"] = node["@timestamp"]
397 
398  tags = dict()
399  if "tag" in node:
400  for tag in node["tag"]:
401  if type(tag) == dict:
402  tags[tag["@k"]] = tag["@v"].strip()
403  # continue
404  else:
405  tags[node['tag']["@k"]] = node['tag']["@v"].strip()
406  # continue
