41 def createExtension(icon):
42  """
43  Create an extended feature with Osmand styling
44  """
45  # camp_pitch.png, tourism_camp_site.png, topo_camp_pitch.png, topo_camp_site.png
46  # trailhead.png, tourism_picnic_site.png, tourism_picnic_site.png,
47  # tourism_attraction.png, tourism_information.png, information_board.png,
48  # firepit.png, historic_ruins.png, amenity_drinking_water.png,
49  # amenity_toilets.png, amenity_parking.png, special_trekking
50  colors = {'tourism_camp_site': '#ff5020', 'tourism_picnic_site': '#ff5020', 'special_trekking': '#a71de1'}
51  nsmap = {'osmand': 'https://osmand.net'}
52  png = etree.Element('{osmand}icon', nsmap=nsmap)
53  png.text = icon
54  back = etree.Element('{osmand}background', nsmap=nsmap)
55  back.text = "circle"
56  color = None
57  if icon in colors:
58  color = etree.Element('{osmand}color', nsmap=nsmap)
59  color.text = colors[icon]
60  return (png, back, color)
61  else:
62  return (png, back)
