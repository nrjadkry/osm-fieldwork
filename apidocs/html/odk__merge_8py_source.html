160  def conflateFile(self,
161  feature: dict,
162  ):
163  """
164  Conflate a POI against all the features in a GeoJson file
165 
166  Args:
167  feature (dict): The feature to conflate
168 
169  Returns:
170  (dict): The modified feature
171  """
172  # Most smartphone GPS are 5-10m off most of the time, plus sometimes
173  # we're standing in front of an amenity and recording that location
174  # instead of in the building.
175  gps_accuracy = 10
176  # this is the treshold for fuzzy string matching
177  match_threshold = 80
178  # log.debug(f"conflateFile({feature})")
179  hits = False
180  data = dict()
181  geom = Point((float(feature["attrs"]["lon"]), float(feature["attrs"]["lat"])))
182  wkt = shape(geom)
183  for existing in self.datadata['features']:
184  id = int(existing['properties']['id'])
185  entry = shapely.from_geojson(str(existing))
186  if entry.geom_type != 'Point':
187  center = shapely.centroid(entry)
188  else:
189  center = entry
190  # dist = shapely.hausdorff_distance(center, wkt)
191  # if 'name' in existing['properties']:
192  # print(f"DIST1: {dist}, {existing['properties']['name']}")
193  # x = shapely.distance(wkt, entry)
194  # haversine reverses the order of lat & lon from what shapely uses. We
195  # use this as meters is easier to deal with than cartesian coordinates.
196  x1 = (center.coords[0][1], center.coords[0][0])
197  x2 = (wkt.coords[0][1], wkt.coords[0][0])
198  dist = haversine(x1, x2, unit=Unit.METERS)
199  if dist < gps_accuracy:
200  # if 'name' in existing['properties']:
201  # log.debug(f"DIST2: {dist}")
202  # log.debug(f"Got a Hit! {feature['tags']['name']}")
203  for key,value in feature['tags'].items():
204  if key in self.analyzeanalyze:
205  if key in existing['properties']:
206  result = fuzz.ratio(value, existing['properties'][key])
207  if result > match_threshold:
208  # log.debug(f"Matched: {result}: {feature['tags']['name']}")
209  existing['properties']['fixme'] = "Probably a duplicate!"
210  log.debug(f"Got a dup in file!!! {existing['properties']['name'] }")
211  hits = True
212  break
213  if hits:
214  version = int(existing['properties']['version'])
215  # coords = feature['geometry']['coordinates']
216  # lat = coords[1]
217  # lon = coords[0]
218  attrs = {'id': id, 'version': version, 'lat': feature['attrs']['lat'], 'lon': feature['attrs']['lon']}
219  tags = existing['properties']
220  tags['fixme'] = "Probably a duplicate!"
221  # Data extracts for ODK Collect
222  del tags['title']
223  del tags['label']
224  if 'building' in tags:
225  return {'attrs': attrs, 'tags': tags, 'refs': list()}
226  return {'attrs': attrs, 'tags': tags}
227  return dict()

465 def conflateThread(features: dict,
466  source: str,
467  dbindex: int,
468  ):
469  """
470  Conflate a subset of the data
471 
472  Args:
473  feature (dict): The feature to conflate
474  source (str): The data source for conflation, file or database
475  dbindex (int): An index into the array of postgres connections
476 
477  Returns:
478  """
479  timer = Timer(text="conflateThread() took {seconds:.0f}s")
480  timer.start()
481  log.debug(f"conflateThread() called! {len(features)} features")
482  merged = list()
483  result = dict()
484  dups = 0
485  # This is brute force, slow but accurate. Process each feature
486  # and look for a possible match with existing data.
487  for key, value in features.items():
488  id = int(value['attrs']['id'])
489  # Each of the conflation methods take a single feature
490  # as a parameter, and returns a possible match or a zero
491  # length dictionary.
492  if id > 0:
493  # Any feature ID greater than zero is existing data.
494  if source.source[:3] != "PG:":
495  result = source.conflateFile(value)
496  else:
497  # Any feature ID less than zero is new data collected
498  # using geopoint in the XLSForm.
499  result = source.conflateById(value, dbindex)
500  elif id < 0:
501  if source.source[:3] != "PG:":
502  result = source.conflateFile(value)
503  else:
504  result = source.conflateNode(value, dbindex)
505  if len(result) == 0:
506  result = source.conflateWay(value, dbindex)
507  if result and len(result) > 0:
508  # Merge the tags and attributes together, the OSM data and ODK data.
509  # If no match is found, the ODK data is used to create a new feature.
510  if 'fixme' in result['tags']:
511  dups += 1
512  # newf = source.cleanFeature(result)
513  attrs = value['attrs'] | result['attrs']
514  tags = value['tags'] | result['tags']
515  merged.append({'attrs': attrs, 'tags': tags})
516  else:
517  merged.append(value)
518  else:
519  merged.append(value)
520  # log.error(f"There are no results!")
521  timer.stop()
522  log.debug(f"Found {dups} duplicates")
523  return merged
