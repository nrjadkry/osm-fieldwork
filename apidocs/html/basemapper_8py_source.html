183  def getTiles(self,
184  zoom: int = None,
185  ):
186  """
187  Get a list of tiles for the specifed zoom level
188 
189  Args:
190  zoom (int): The Zoom level of the desired map tiles
191 
192  Returns:
193  (int): The total number of map tiles downloaded
194  """
195  info = get_cpu_info()
196  cores = info["count"]
197 
198  self.tilestiles = list(
199  mercantile.tiles(
200  self.bboxbbox[0], self.bboxbbox[1], self.bboxbbox[2], self.bboxbbox[3], zoom)
201  )
202  total = len(self.tilestiles)
203  log.info("%d tiles for zoom level %d" % (len(self.tilestiles), zoom))
204  chunk = round(len(self.tilestiles) / cores)
205  queue.Queue(maxsize=cores)
206  log.info("%d threads, %d tiles" % (cores, total))
207 
208  mirrors = [self.sourcessources[self.sourcesource]]
209  # epdb.st()
210  if len(self.tilestiles) < chunk or chunk == 0:
211  dlthread(self.basebase, mirrors, self.tilestiles, self.xyxy)
212  else:
213  with concurrent.futures.ThreadPoolExecutor(max_workers=cores) as executor:
214  block = 0
215  while block <= len(self.tilestiles):
216  executor.submit(
217  dlthread, self.basebase, mirrors, self.tilestiles[block : block + chunk]
218  )
219  log.debug("Dispatching Block %d:%d" % (block, block + chunk))
220  block += chunk
221  executor.shutdown()
222  # log.info("Had %r errors downloading %d tiles for data for %r" % (self.errors, len(tiles), os.path.basename(self.base)))
223 
224  return len(self.tilestiles)

255 
256  Returns:

260  # Is BBOX string
261  try:

263  bbox = tuple(float(x) for x in bbox_parts)
264  if len(bbox) == 4:
265  # BBOX valid
266  return bbox
267  else:
268  log.error(f"BBOX string malformed: {bbox}")
269  return
270  except Exception as e:

272  log.error(f"Failed to parse BBOX string: {boundary}")
273  return
274 
275  log.debug(f"Reading geojson file: {boundary}")
276  with open(boundary, "r") as f:
277  poly = json.load(f)
278  if 'features' in poly:
279  geometry = shape(poly['features'][0]['geometry'])
280  else:
281  geometry = shape(poly)
282 
283  if isinstance(geometry, list):
284  # Multiple geometries
285  log.debug("Creating union of multiple bbox geoms")
286  geometry = unary_union(geometry)
287 
288  if geometry.is_empty:
289  log.warning(f"No bbox extracted from {geometry}")
290  return None
291 
292  bbox = geometry.bounds
293  # left, bottom, right, top
294  # minX, minY, maxX, maxY
295  return bbox
