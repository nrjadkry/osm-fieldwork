184  """
185  Parse the JSON file from ODK Central and convert it to a data structure.
186  The input is either a filespec to open, or the data itself.
187 
188  Args:
189  filespec (str): The JSON or GeoJson input file to convert
190  data (str): The data to convert
191 
192  Returns:
193  (list): A list of all the features in the input file
194  """
195  log.debug(f"Parsing JSON file {filespec}")
196  all_tags = list()
197  if not data:
198  file = open(filespec, "r")
199  infile = Path(filespec)
200  if infile.suffix == ".geojson":
201  reader = geojson.load(file)
202  elif infile.suffix == ".json":
203  reader = json.load(file)
204  else:
205  log.error("Need to specify a JSON or GeoJson file!")
206  return all_tags
207  elif isinstance(data, str):
208  reader = geojson.loads(data)
209  elif isinstance(data, list):
210  reader = data
211 
212  total = list()
213  # JSON files from Central use value as the keyword, whereas
214  # GeoJSON uses features for the same thing.
215  if 'value' in reader:
216  data = reader['value']
217  elif 'features' in reader:
218  data = reader['features']
219  else:
220  data = reader
221  for row in data:
222  # log.info(f"ROW: {row}")
223  tags = dict()
224  if 'geometry' in row:
225  # If geom not point, convert to centroid
226  if row['geometry']['type'] != 'Point':
227  log.debug(f"Converting {row['geometry']['type']} geometry to centroid point")
228  geom = shapely.from_geojson(str(row))
229  centroid = shapely.to_geojson(geom.centroid)
230  row['geometry'] = centroid
231  tags['geometry'] = row['geometry']
232  else:
233  pat = re.compile("[-0-9.]*, [0-9.-]*, [0-9.]*")
234  gps = re.findall(pat, str(row))
235  # If geopoint warmup is used, there will be two matches, we only
236  # want the second one, which is the location.
237  for coords in gps:
238  tags['geometry'] = coords
239  if 'properties' in row:
240  indata = row['properties'] # A GeoJson formatted file
241  else:
242  indata = row # A JOSM file from ODK Central
243 
244  # flatten all the groups into a single data structure
245  flattened = flatdict.FlatDict(row)
246  for k, v in flattened.items():
247  last = k.rfind(':') + 1
248  key = k[last:]
249  # log.debug(f"Processing tag {key} = {v}")
250  # names and comments may have spaces, otherwise
251  # it's from a select_multiple
252  pat = re.compile("name[:a-z]*")
253  names = re.findall(pat, key)
254  if len(names) > 0:
255  for name in names:
256  tags[name] = v
257  continue
258  if key == 'comment':
259  tags[key] = v
260  # a JSON file from ODK Central always uses coordinates as
261  # the keyword
262  if key == 'coordinates':
263  if isinstance(v, list):
264  lat = v[1]
265  lon = v[0]
266  tags['geometry'] = f"{lat} {lon}"
267  continue
268  if key == "xlocation":
269  tags["geometry"] = v
270  continue
271  tags[key] = v
272  total.append(tags)
273 
274  log.debug(f"Finsished parsing JSON file {filespec}")
275  return total

385 def json2osm(input_file, yaml_file=None):
386  """
387  Process the JSON file from ODK Central or the GeoJSON file to OSM XML format.
388 
389  Args:
390  input_file (str): The path to the input JSON or GeoJSON file.
391  yaml_file (str): The path to the YAML config file (optional).
392 
393  Returns:
394  osmoutfile (str): Path to the converted OSM XML file.
395  """
396  log.info(f"Converting JSON file to OSM: {input_file}")
397  if yaml_file:
398  jsonin = JsonDump(yaml_file)
399  else:
400  jsonin = JsonDump()
401 
402  # jsonin.parseXLS(args.xlsfile)
403 
404  # Modify the input file name for the 2 output files, which will get written
405  # to the current directory.
406 
407  base = Path(input_file).stem
408  osmoutfile = f"{base}-out.osm"
409  jsonin.createOSM(osmoutfile)
410 
411  data = jsonin.parse(input_file)
412  # This OSM XML file only has OSM appropriate tags and values
413 
414  for entry in data:
415  feature = jsonin.createEntry(entry)
416 
417  # Sometimes bad entries, usually from debugging XForm design, sneak in
418  if len(feature) == 0:
419  continue
420 
421  if len(feature) > 0:
422  if "lat" not in feature["attrs"]:
423  if 'geometry' in feature['tags']:
424  if isinstance(feature["tags"]["geometry"], str):
425  coords = list(feature['tags']['geometry'])
426  # del feature['tags']['geometry']
427  elif 'coordinates' in feature['tags']:
428  coords = feature['tags']['coordinates']
429  feature['attrs'] = {'lat': coords[1], 'lon': coords[0]}
430  else:
431  log.warning(f"Bad record! {feature}")
432  continue # Skip bad records
433 
434  log.debug("Writing final OSM XML file...")
435  jsonin.writeOSM(feature)
436 
437  jsonin.finishOSM()
438  log.info(f"Wrote OSM XML file: {osmoutfile}")
439
